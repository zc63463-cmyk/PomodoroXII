# PomodoroXII 深度架构规划

> 生成时间：2026-07-01 | 基于 React 19 技术栈重构

---

## 一、设计哲学

### 1.1 三条铁律

1. **Routers commit, Services flush** — 事务控制权在请求边界，Service 只 flush 不 commit
2. **Services 不导入 FastAPI** — 业务逻辑与 HTTP 解耦，可独立测试
3. **Models 纯数据** — ORM 模型只有字段和约束，无业务方法

### 1.2 从 pomodoroxi 移植什么、重构什么、新建什么

| 类别 | 具体内容 | 来源 |
|------|---------|------|
| **移植（原样搬）** | Outbox 合并矩阵、多层 Tombstone 防复活、零点时间戳检测、ENTITY_REGISTRY 模式、Cloudflare 重试逻辑、Dexie schema v15 | pomodoroxi 战斗经验 |
| **移植（改语言）** | file_system 15 文件子系统（8 Mixin + 三层锁 + FTS5） | video4KB |
| **重构（拆解重组）** | sync.py 1047 行 → SyncService + 安全检查函数 + adapter | pomodoroxi |
| **重构（提抽取）** | 级联删除逻辑 → CascadeService（REST + sync 共用） | pomodoroxi |
| **新建** | Alembic 迁移、结构化日志、异常体系、MCP Server、Export API、定时快照、.md frontmatter | 全新 |

### 1.3 参考的成熟项目

| 项目 | 借鉴点 | 参考方式 |
|------|--------|---------|
| **FastAPI 官方生产模式** | 三层架构、命名约定、lifespan、结构化日志 | 项目结构 |
| **Joplin** | REST API 资源路由 + 事件流 `/events` + Token 认证 | API 设计 |
| **Obsidian** | .md frontmatter 自描述 + diff-match-patch 冲突合并 | 文件格式 |
| **Tolaria** | AGENTS.md AI 导航 + Files-first 可移植性 | Agent 友好 |
| **MCP 协议** | Tools + Resources + Prompts 三概念 | Agent 接入 |
| **PowerSync** | Bucket 分桶同步 + Write Checkpoints | 同步引擎（长期） |

```python
# app/services/base.py
class BaseService:
    def __init__(self, db: AsyncSession):
        self.db = db

    async def get(self, model_cls, entity_id: str):
        obj = await self.db.get(model_cls, entity_id)
        if not obj:
            raise NotFoundError(model_cls.__name__, entity_id)
        return obj

    async def create(self, model_cls, data: dict):
        obj = model_cls(**data)
        obj.version = 1
        self.db.add(obj)
        await self.db.flush()
        return obj

    async def update(self, obj, data: dict):
        for key, value in data.items():
            setattr(obj, key, value)
        obj.version += 1
        await self.db.flush()
        return obj

# app/services/note_service.py — Saga 模式桥接 DB + FS
class NoteService(BaseService):
    def __init__(self, db: AsyncSession, fs: FileSystemStorage):
        super().__init__(db)
        self.fs = fs

    async def create_note(self, data: dict, content: str) -> Note:
        note_id = data["id"]
        try:
            note = await self.create(Note, data)
            await self.fs.create_note(
                title=data.get("title", ""),
                content=content,
                folder_id=data.get("folder_id"),
                external_id=note_id,
            )
            note.content_hash = _sha256(content)
            note.word_count = len(content.split())
            await self.db.flush()
            return note
        except Exception:
            await self.db.rollback()
            try:
                await self.fs.delete_note_permanent(note_id)
            except Exception:
                logger.error(f"Saga compensation failed for note {note_id}")
            raise

    async def update_note_content(self, note_id: str, content: str) -> Note:
        note = await self.get(Note, note_id)
        old_hash = note.content_hash
        try:
            await self.fs.edit_note(note_id, content)
            note.content_hash = _sha256(content)
            note.word_count = len(content.split())
            await self.db.flush()
            return note
        except Exception:
            note.content_hash = old_hash  # 恢复
            await self.db.rollback()
            raise

    async def delete_note(self, note_id: str, create_tombstone: bool = True):
        note = await self.get(Note, note_id)
        await self.fs.delete_note(note_id)
        note.trashed_at = utc_now_iso()
        note.status = "trashed"
        if create_tombstone:
            await TombstoneService(self.db).create("note", note_id)
        await self.db.flush()
```

```python
# app/services/sync_safety.py — 从 sync.py 提取的可复用安全检查
async def check_tombstone_first(db, entity_type, entity_id) -> dict | None:
    """墓碑优先检查 — 防止已删除实体被 push 复活"""
    tomb = await db.execute(
        select(Tombstone).where(
            Tombstone.entity_type == entity_type,
            Tombstone.entity_id == entity_id,
        )
    )
    if tomb.scalar_one_or_none():
        return {"conflict": {"reason": "tombstone_exists"}}
    return None

def strip_client_fields(entity_data: dict, entity_type: str) -> dict:
    """剥离客户端独有字段"""
    for key in ("synced", "_dirty", "_etag"):
        entity_data.pop(key, None)
    if entity_type == "task":
        entity_data.pop("actual_pomodoros", None)
    if entity_type == "quickNote":
        entity_data.pop("archive_file_path", None)
        entity_data.pop("migrated_to_note_id", None)
    return entity_data

def detect_zero_time(ts: str | None) -> bool:
    """零点时间戳检测 — 外部脚本注入的陈旧数据"""
    if not ts:
        return False
    time_part = ts.split("T")[-1] if "T" in ts else ts
    return time_part.startswith("00:00:00")

async def check_folder_circular_ref(db, folder_id, new_parent_id) -> bool:
    """文件夹循环引用检测"""
    if folder_id == new_parent_id:
        return True
    queue = [new_parent_id]
    while queue:
        current = queue.pop()
        children = await db.execute(select(Folder.id).where(Folder.parent_id == current))
        for (child_id,) in children.all():
            if child_id == folder_id:
                return True
            queue.append(child_id)
    return False

# app/services/sync_service.py — 精简后的 sync 引擎
class SyncService:
    def __init__(self, db: AsyncSession, fs: FileSystemStorage):
        self.db = db
        self.fs = fs

    async def push(self, client_id: str, events: list[dict]) -> dict:
        applied, conflicts, errors = [], [], []
        for idx, event in enumerate(events):
            event_applied = False
            try:
                async with self.db.begin_nested():  # SAVEPOINT per event
                    event_applied = await self._handle_event(event)
                if event_applied:
                    applied.append(idx)
            except Exception as exc:
                errors.append({"index": idx, "error": str(exc)})
                continue
        try:
            await self.db.commit()  # 循环外统一 commit
        except Exception:
            await self.db.rollback()
            return {"applied": [], "conflicts": [], "errors": [...]}
        return {"applied": applied, "conflicts": conflicts, "errors": errors}

    async def _handle_event(self, event: dict) -> bool:
        etype = event["type"]
        action = event["action"]
        entity = strip_client_fields(event.get("entity", {}).copy(), etype)
        entity_id = event["entity_id"]
        client_time = event["client_time"]

        if detect_zero_time(entity.get("updated_at")):
            entity["updated_at"] = utc_now_iso()

        if action in ("create", "update"):
            tomb_check = await check_tombstone_first(self.db, etype, entity_id)
            if tomb_check:
                return False

        if etype in ("note", "folder"):
            return await self._handle_fs_entity(etype, action, entity_id, entity, client_time)
        else:
            return await self._handle_orm_entity(etype, action, entity_id, entity, client_time)
```

```python
# app/services/cascade_service.py — REST 和 sync 共用
class CascadeService:
    def __init__(self, db: AsyncSession, fs: FileSystemStorage | None = None):
        self.db = db
        self.fs = fs

    async def delete_task_cascade(self, task_id: str):
        """删除 task + 关联 sessions + 清理 quicknote 引用 + tombstone"""
        now = utc_now_iso()
        # 级联删除 sessions
        sessions = await self.db.execute(
            select(PomodoroSession).where(PomodoroSession.task_id == task_id)
        )
        for s in sessions.scalars().all():
            await self.db.execute(
                update(QuickNote).where(QuickNote.session_id == s.id)
                .values(session_id=None, updated_at=now)
            )
            await self.db.execute(delete(PomodoroSession).where(PomodoroSession.id == s.id))
            await TombstoneService(self.db).create("session", s.id)
        # 删除关联表
        await self.db.execute(delete(TaskQuickNote).where(TaskQuickNote.task_id == task_id))
        # tombstone
        await TombstoneService(self.db).create("task", task_id)
        # 物理删除
        await self.db.execute(delete(Task).where(Task.id == task_id))

    async def delete_folder_cascade(self, folder_id: str):
        """BFS 收集后代 + 级联删除 + 清理 Note/QuickNote folder_id"""
        # BFS 收集后代
        descendant_ids = await self._bfs_descendants(folder_id)
        for desc_id in descendant_ids + [folder_id]:
            await self.db.execute(
                update(Note).where(Note.folder_id == desc_id)
                .values(folder_id=None, updated_at=utc_now_iso())
            )
            await self.db.execute(
                update(QuickNote).where(QuickNote.folder_id == desc_id)
                .values(folder_id=None, updated_at=utc_now_iso())
            )
            await self.db.execute(delete(Folder).where(Folder.id == desc_id))
            await TombstoneService(self.db).create("folder", desc_id)

    async def delete_note_cascade(self, note_id: str):
        """删除 note + FS 软删除 + tombstone"""
        if self.fs:
            await self.fs.delete_note(note_id)
        await self.db.execute(delete(Note).where(Note.id == note_id))
        await TombstoneService(self.db).create("note", note_id)
```

```python
import yaml

def build_md_content(meta: dict, content: str) -> str:
    """构建带 YAML frontmatter 的 .md 文件"""
    frontmatter = {
        "id": meta["note_id"],
        "title": meta["title"],
        "tags": meta.get("tags", []),
        "folder_id": meta.get("folder_id"),
        "content_hash": meta.get("content_hash", ""),

        "created_at": _to_iso_z(meta.get("created_at")),
        "updated_at": _to_iso_z(meta.get("updated_at")),
    }
    return f"---\n{yaml.dump(frontmatter, allow_unicode=True, default_flow_style=False)}---\n\n{content}"

def parse_frontmatter(md_text: str) -> tuple[dict, str]:
    """解析 .md 文件，返回 (frontmatter_dict, body)"""
    if not md_text.startswith("---\n"):
        return {}, md_text
    parts = md_text[4:].split("\n---\n", 1)
    if len(parts) != 2:
        return {}, md_text
    return yaml.safe_load(parts[0]), parts[1]
```
