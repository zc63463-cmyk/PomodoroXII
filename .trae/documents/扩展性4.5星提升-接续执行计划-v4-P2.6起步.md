```python
#!/usr/bin/env python
"""scaffold_entity.py — 生成新实体的模型/Schema/Service/Router/Registry 草案。

MVP: 只支持 DB-only + hard_tombstone 实体。
Note/Folder/Relation 类特殊实体请用 --custom-service 标记后手工实现。
"""
from __future__ import annotations

import argparse
import sys
from pathlib import Path

TYPE_MAP = {
    "string": "String",
    "text": "Text",
    "integer": "Integer",
    "float": "Float",
    "boolean": "Boolean",
    "datetime": "DateTime",
    "json": "JSON",
}


def parse_fields(raw: str) -> list[tuple[str, str]]:
    """解析 'title:string,due_date:datetime' 为 [(name, type), ...]。"""
    result = []
    for chunk in raw.split(","):
        chunk = chunk.strip()
        if not chunk:
            continue
        if ":" not in chunk:
            raise ValueError(f"Invalid field spec: {chunk!r} (expected name:type)")
        name, ftype = chunk.split(":", 1)
        name, ftype = name.strip(), ftype.strip()
        if ftype not in TYPE_MAP:
            raise ValueError(f"Unknown field type: {ftype!r} (valid: {list(TYPE_MAP)})")
        result.append((name, ftype))
    return result


def render_model(name: str, class_name: str, table_name: str, fields: list[tuple[str, str]]) -> str:
    """渲染 ORM 模型文件。"""
    cols = []
    for fname, ftype in fields:
        col_type = TYPE_MAP[ftype]
        cols.append(f"    {fname} = Column({col_type}, nullable=False)")
    cols_str = "\n".join(cols)
    return f'''"""{class_name} model."""
from __future__ import annotations

from sqlalchemy import Column
from sqlalchemy.orm import Mapped, mapped_column

from app.db.base import Base
from app.models.mixins import SyncMixin


class {class_name}(SyncMixin, Base):
    """TODO: describe {name}."""
    __tablename__ = "{table_name}"

{cols_str}
'''

def main():
    parser = argparse.ArgumentParser(description="Scaffold a new entity")
    parser.add_argument("--name", required=True, help="entity name (snake_case)")
    parser.add_argument("--class-name", required=True, help="ORM class name (PascalCase)")
    parser.add_argument("--table-name", required=True, help="SQL table name")
    parser.add_argument("--route-prefix", required=True, help="REST route prefix")
    parser.add_argument("--fields", required=True, help='Comma-separated fields: "title:string,due_date:datetime"')
    parser.add_argument("--dry-run", action="store_true", help="Print to stdout instead of writing files")
    args = parser.parse_args()

    fields = parse_fields(args.fields)

    blocks = [
        (f"app/models/{args.name}.py", render_model(args.name, args.class_name, args.table_name, fields)),
        (f"app/schemas/{args.name}.py", render_schema(args.name, args.class_name, fields)),

            print(content)
            print()

                print(content, file=sys.stderr)
                continue
            target = Path(path)
            target.parent.mkdir(parents=True, exist_ok=True)
            target.write_text(content, encoding="utf-8")
            print(f"已生成: {path}")


if __name__ == "__main__":
```

```python
"""Parity gate: REST /stats 端点 vs MCP stats 工具。

用 STAT_SPECS 作为单一事实源,确保每个 REST stats 端点有对应 MCP 工具(或显式豁免)。
"""
from __future__ import annotations

import inspect

from app.services.stats_spec import STAT_SPECS


def _get_mcp_tool_names() -> set[str]:
    """从 app.mcp.server 模块收集所有 @mcp.tool 装饰的函数名。"""
    import app.mcp.server as mcp_module
    names: set[str] = set()
    for name, obj in inspect.getmembers(mcp_module):
        if callable(obj) and not name.startswith("_"):
            if name.startswith("get_") or name.startswith("list_") or name.startswith("sync_"):
                names.add(name)
    return names


def test_mcp_has_all_stats_tools():
    """MCP server 必须注册所有 STAT_SPECS 中 mcp_enabled=True 的工具。"""
    tool_names = _get_mcp_tool_names()
    expected = {spec.mcp_tool for spec in STAT_SPECS if spec.mcp_enabled}
    missing = expected - tool_names
    assert not missing, f"MCP missing stats tools: {missing}"


def test_stat_specs_covers_all_rest_endpoints():
    """STAT_SPECS 必须覆盖所有 REST /stats 端点(反向校验)。"""
    rest_paths = {s.route_path for s in STAT_SPECS}
    expected = {
```
